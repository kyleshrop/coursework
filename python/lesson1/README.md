Running Flask in Venv:
    #create venv file in app folder
    <python3 -m venv venv>

    #activate venv environment
    <. venv/bin/activate>

    #Flask installation
    <pip install Flask>

    #Export app {main.py}
    <export FLASK_APP={main}>

    #Use flask to run the app
    <flask run>
